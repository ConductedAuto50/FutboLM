import matplotlib.pyplot as plt
import numpy as np

seasons = ['2002-2003', '2003-2004', '2004-2005', '2005-2006', '2006-2007', '2007-2008', '2008-2009', '2009-2010', '2010-2011', '2011-2012', '2012-2013', '2013-2014', '2014-2015', '2015-2016', '2016-2017', '2017-2018', '2018-2019', '2019-2020', '2020-2021', '2021-2022', '2022-2023', '2023-2024', '2024-2025']
goals = [3, 4, 5, 9, 20, 39, 22, 33, 46, 56, 46, 48, 61, 51, 38, 42, 28, 37, 36, 24, 17, 35, 23]

x = np.arange(len(seasons))

plt.figure(figsize=(15, 7))
plt.bar(x, goals)
plt.xlabel('Season')
plt.ylabel('Goals')
plt.title('Cristiano Ronaldo Goals Per Season (All Competitions)')
plt.xticks(x, seasons, rotation=45, ha='right')
plt.tight_layout()
plt.savefig('graph.png')
